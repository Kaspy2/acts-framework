if(USE_HEPMC)
  file(GLOB src_files "src/HepMC.cpp" "include/ACTFW/Plugins/HepMC/HepMC.hpp")
  add_library(ACTFWPluginHepMC SHARED ${src_files})
  target_include_directories(ACTFWPluginHepMC PUBLIC $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include/> $<INSTALL_INTERFACE:include>)
  target_link_libraries(ACTFWPluginHepMC PUBLIC ACTSCore ACTFramework)
  target_link_libraries(ACTFWPluginHepMC PRIVATE HepMCLib) 
  target_link_libraries(ACTFWPluginHepMC PRIVATE HepPID)

  install(TARGETS ACTFWPluginHepMC LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR})
  install(DIRECTORY include/ACTFW DESTINATION ${CMAKE_INSTALL_INCLUDEDIR})
endif()
